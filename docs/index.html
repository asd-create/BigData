hadoop完全分布式系统

       1.准备 3 台客户机（关闭防火墙、配置静态 ip、主机名称、改hosts）

              (1).关闭防火墙

                     systemctl stop firewalld //关闭防火墙

                     systemctl disable firewalld   //开机关闭防火墙

              (2).改静态ip

                     vim /etc/sysconfig/network-scripts/ifcfg-ens33

                    

                     TYPE="Ethernet"

                     BOOTPROTO="static"

                     NAME="ens33"

                     DEVICE="ens33"

                     ONBOOT="yes"

                     PREFIX=24

                     IPADDR=192.168.121.100

                     GATEWAY=192.168.121.2                        //要与NAT设置中的网关一致

                     DNS1=192.168.121.2

                     备注：设置好记得刷新service network restart

              (3).主机名称

                     vim /etc/hostname

                    

                     hadoop101

                     备注：跟随主机而变

              (4).改hosts

                     vim /etc/hosts

                    

                     192.168.121.100 hadoop100

                     192.168.121.101 hadoop101

                     192.168.121.102 hadoop102

                     192.168.121.103 hadoop103

                     备注：克隆的时候记得添加数据                   

       2.上传并解压jdk

              (1).检查jdk是否已安装并卸载

                     rpm -qa | grep jdk

                     rpm -e --node +包名

              (2).解压jdk

                     tar -zxvf jdk-8u141-linux-x64.tar.gz -C /opt/module/

              (3).配置jdk环境变量

                     vim /etc/profile

                    

                     #JAVA_HOME

                     export JAVA_HOME=/opt/module/jdk1.8.0_131

                     export PATH=$PATH:$JAVA_HOME/bin

                     备注：export指提升为全局变量

                              source /etc/profile  让该文件生效

              (4).检查是否安装成功

                     java -version

       3.上传并解压hadoop

              (1).检查jdk是否已安装并卸载

                     rpm -qa | grep jdk

                     rpm -e --node +包名

              (2).解压hadoop

                     tar -zxvf hadoop-2.6.0.tar.gz -C /opt/module/

              (3).配置hadoop的环境变量

                     vim /etc/profile

                    

                     #HADOOP_HOME

                     export HADOOP_HOME=/opt/module/hadoop-2.6.0

                     export PATH=$PATH:$HADOOP_HOME/bin:$HADOOP_HOME:/sbin

                     备注：source /etc/profile  让该文件生效

       4.配置hadoop的集群相关信息

              (1).配置hadoop-env.sh、mapred-env.sh、yarn-env.sh

                     vim etc/hadoop/hadoop-env.sh

                    

                     export JAVA_HOME=/opt/module/jdk1.8.0_131

              (2).配置core-site.xm、hdfs-site.xml、mapred-site.xml、yarn-site.xml

                     1. vim core-site.xm

                            <configuration>

                           

                            <!-- 指定 HDFS 中 NameNode 的地址 -->

                            <property>

                            <name>fs.defaultFS</name>

                            <value>hdfs://hadoop102:9000</value>

                            </property>

                           

                            <!-- 指定 Hadoop 运行时产生文件的存储目录 -->

                            <property>

                            <name>hadoop.tmp.dir</name>

                            <value>/opt/module/hadoop-2.6.0/data/tmp</value>

                            </property>

                           

                            </configuration>

                     2.vim hdfs-site.xml

                            <configuration>

                           

                            <!-- 指定 HDFS 副本的数量 -->

                            <property>

                            <name>dfs.replication</name>

                            <value>1</value>

                            </property>

                           

                            <!-- name的目录 -->

                            <property>

                            <name>dfs.namenode.name.dir</name>

                            <value>/opt/module/hadoop-2.6.0/data/tmp/dfs/name</value>

                            </property>

                           

                            <!-- 指定 Hadoop 辅助名称节点主机配置 -->

                            <property>

                            <name>dfs.namenode.secondary.http-address</name>

                            <value>hadoop104:50090</value>

                            </property>

                           

                            <!-- data的目录 -->

                            <property>

                            <name>dfs.datanode.data.dir</name>

                            <value>/opt/module/hadoop-2.6.0/data/tmp/dfs/data</value>

                            </property>

                           

                            </configuration>

                     3.vim mapred-site.xml

                            <configuration>

                           

                            <!-- 指定 MR 运行在 YARN 上 -->

                            <property>

                            <name>mapreduce.framework.name</name>

                            <value>yarn</value>

                            </property>

                           

                            <!-- 历史服务器端地址 -->

                            <property>

                            <name>mapreduce.jobhistory.address</name>

                            <value>hadoop102:10020</value>

                            </property>

                           

                            <!-- 历史服务器 web 端地址 -->

                            <property>

                            <name>mapreduce.jobhistory.webapp.address</name>

                            <value>hadoop102:19888</value>

                            </property>

                    

                            </configuration>

                     4.vim yarn-site.xml

                            <configuration>

                           

                            <!-- Reducer 获取数据的方式 -->

                            <property>

                            <name>yarn.nodemanager.aux-services</name>

                            <value>mapreduce_shuffle</value>

                            </property>

                           

                            <!-- 指定 YARN 的 ResourceManager 的地址 -->

                            <property>

                            <name>yarn.resourcemanager.hostname</name>

                            <value>hadoop103</value>

                            </property>

                           

                            <!-- 日志聚集功能使能 -->

                            <property>

                            <name>yarn.log-aggregation-enable</name>

                            <value>true</value>

                            </property>

                           

                            <!-- 日志保留时间设置 7 天 -->

                            <property>

                            <name>yarn.log-aggregation.retain-seconds</name>

                            <value>604800</value>

                            </property>

                           

                            </configuration>
